void deleteTankById(TankList *list, int id){
    if(list == NULL)
        exit(EXIT_FAILURE);
    Tank *first = list->firstTank;
    Tank *toDelete;
    while(first != NULL){
        if(first->id == id){
            toDelete = first;
            list->firstTank = list->firstTank->next;
            free(toDelete);
            break;
        }
        first = first->next;
    }
}

    }
